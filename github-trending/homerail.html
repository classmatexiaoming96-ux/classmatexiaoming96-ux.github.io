HomeRail：把本地 coding agent 变成可审计 DAG 工作流

今天筛到的 HomeRail 不是“又一个 Claude Code prompt 包”，而是一个把工作流编译、运行时调度、worker WebSocket、容器化隔离和 Agent UI 串起来的本地编排系统。源码里最值得看的不是 UI，而是它如何把 agent 的不确定输出约束成可恢复、可验证、可路由的 DAG。

 261    run: DAGRun,
 262    fromNode: string,
 263    port: string,
 264    content: unknown,
 265  ): DAGTransitionResult {
 266    if (!run.nodeStates.has(fromNode)) {
 267      throw new Error(`Unknown node: ${fromNode}`);
 268    }
 269  
 270    const matchingDownstream = run.graph.edges.filter(
 271      (edge) =>
 272        edge.from_node === fromNode &&
 273        edge.label !== "after_dep" &&
 274        edge.to_node !== "" &&
 275        edgeMatchesHandoff(edge, port),
 276    );
 277    const matchingTerminal = run.graph.edges.find(
 278      (edge) => edge.from_node === fromNode && edge.to_node === "" && edgeMatchesHandoff(edge, port),
 279    );
 280    const terminalOutcome = matchingTerminal?.terminal_outcome;
 281    const terminalFailure = terminalOutcome === "failure" ||
 282      (terminalOutcome === undefined && isFailurePort(port) && matchingDownstream.length === 0);
 283    const terminalCancelled = terminalOutcome === "cancelled";
 284    run.handoffedNodes.add(fromNode);
 285    run.nodeStates.set(
 286      fromNode,
 287      terminalFailure
 288        ? "FAILED"
 289        : terminalCancelled
 290          ? "CANCELLED"
 291        : run.loopSources.has(fromNode)
 292          ? "RUNNING"
 293          : "COMPLETED",
 294    );
 295  
 296    const affected = terminalFailure ? new Set<string>() : _satisfyAfterDeps(run, fromNode);
 297    const mailboxReceivers = new Set<string>();
 298    for (const edge of matchingDownstream) {
 299      _ensureMailbox(run, edge.to_node, edge.to_port).push(content);
 300      run.inputSatisfied.get(edge.to_node)?.add(fromNode);
 301      affected.add(edge.to_node);
 302      mailboxReceivers.add(edge.to_node);
 303    }
 304  
 305    for (const nodeId of affected) {
 306      _tryPromote(run, nodeId);
 307      _skipUntakenSatisfiedBranches(run, nodeId);
 308    }
 309    for (const nodeId of mailboxReceivers) {
 310      _wakeLoopSource(run, nodeId);
 311      _wakeLoopGatewayReceiver(run, fromNode, nodeId);
 312    }
 313    if (terminalFailure) {
 314      _skipDependentNodes(run, fromNode);
 315    }
 316    return {
 317      affectedNodes: Array.from(affected).sort(),
 318      routedNodes: Array.from(mailboxReceivers).sort(),
