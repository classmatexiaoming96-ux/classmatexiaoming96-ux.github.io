30function writeProject(doc) {
31  // atomic tmp+rename so the UI's file watcher never sees a half-written doc
32  const tmp = PROJECT_FILE + ".mcp.tmp";

34  fs.renameSync(tmp, PROJECT_FILE);

36/* Optimistic concurrency: revision of project.json when this session last read

40function httpOk(url) {
41  return new Promise((resolve) => {
42    const req = http.get(url, (r) => { r.resume(); resolve(r.statusCode < 500); });
43    req.on("error", () => resolve(false));
44    req.setTimeout(1200, () => { req.destroy(); resolve(false); });
45  });

47async function ensureUIServer() {
48  if (await httpOk(BASE + "/api/project")) return true;

▸patch ops 的真实语义

patch_project 会读当前项目、遍历 ops、对 media/clips/project 做局部修改，最后 revision++ 写回。patch_project 分支见 sources/fablecut/mcp-server.js:238；addClip 需要 id/kind/mediaId见 sources/fablecut/mcp-server.js:247；updateClip 合并 props见 sources/fablecut/mcp-server.js:261；removeClip 过滤数组见 sources/fablecut/mcp-server.js:281；写回并返回 project见 sources/fablecut/mcp-server.js:314。

sources/fablecut/mcp-server.js:238-317JavaScript
238    case "fablecut_patch_project": {
239      const ops = args.ops;
240      if (!Array.isArray(ops) || !ops.length) throw new Error("`ops` must be a non-empty array");
241      const proj = readProject();
242      const notes = [];
243      const mergeInto = (target, set) => {
244        for (const [k, v] of Object.entries(set || {})) {
245          if (v === null) delete target[k];
246          else if (k === "props" && target.props && typeof v === "object" && !Array.isArray(v)) {
247            for (const [pk, pv] of Object.entries(v)) {
248              if (pv === null) delete target.props[pk]; else target.props[pk] = pv;
249            }
250          } else target[k] = v;
251        }
252      };
253      for (const op of ops) {
254        switch (op.op) {
255          case "addClip": {
256            const c = op.clip;
257            if (!c || !c.track || typeof c.start !== "number" || typeof c.duration !== "number")
258              throw new Error("addClip needs clip{track, start, duration}");
259            c.id = c.id || "c_" + uid();
260            if (proj.clips.some((x) => x.id === c.id)) throw new Error("addClip: duplicate clip id " + c.id);
261            if (c.kind !== "text" && c.kind !== "adjust" && !proj.media.some((m) => m.id === c.mediaId))
262              throw new Error(`addClip: unknown mediaId ${c.mediaId}`);
263            proj.clips.push(c);
264            notes.push("+" + c.id);
265            break;
266          }
267          case "updateClip": {
268            const c = proj.clips.find((x) => x.id === op.id);
269            if (!c) throw new Error("updateClip: no clip " + op.id);
270            mergeInto(c, op.set);
271            notes.push("~" + op.id);
272            break;
273          }
274          case "removeClip": {
275            const n = proj.clips.length;
276            proj.clips = proj.clips.filter((x) => x.id !== op.id);
277            if (proj.clips.length === n) throw new Error("removeClip: no clip " + op.id);
278            notes.push("-" + op.id);
279            break;
280          }
281          case "addMedia": {
282            const m = op.media;
283            if (!m || !m.src || !m.kind) throw new Error("addMedia needs media{src, kind}");
284            m.id = m.id || "m_" + uid();
285            if (proj.media.some((x) => x.id === m.id)) throw new Error("addMedia: duplicate media id " + m.id);
286            m.name = m.name || path.basename(decodeURIComponent(m.src));
287            proj.media.push(m);
288            notes.push("+" + m.id);
289            break;
290          }
291          case "removeMedia": {
292            const used = proj.clips.find((c) => c.mediaId === op.id);
293            if (used) throw new Error(`removeMedia: media ${op.id} is used by clip ${used.id}`);
294            const n = proj.media.length;
295            proj.media = proj.media.filter((x) => x.id !== op.id);
296            if (proj.media.length === n) throw new Error("removeMedia: no media " + op.id);
297            notes.push("-" + op.id);
298            break;
299          }
300          case "setProject": {
301            const allowed = ["name", "width", "height", "fps", "background", "markers"];
302            for (const [k, v] of Object.entries(op.set || {})) {
303              if (!allowed.includes(k)) throw new Error(`setProject: '${k}' not settable (allowed: ${allowed.join(", ")})`);
304              if (v === null) delete proj[k]; else proj[k] = v;
305            }
306            notes.push("~project");
307            break;
308          }
309          default:
310            throw new Error("Unknown op: " + op.op + " (addClip|updateClip|removeClip|addMedia|removeMedia|setProject)");
311        }
312      }
313      proj.revision = (proj.revision || 0) + 1;
314      writeProject(proj);
315      lastReadRevision = proj.revision;
316      return `Patched (revision ${proj.revision}): ${notes.join(" ")}. Now ${proj.clips.length} clip(s), ${proj.media.length} media. UI hot-reloaded.`;
317    }
§ 02 · KEY FLOW

关键流程

Agent calls fablecut_status → compact project summary │ ├─ fablecut_import_media(localPath) → copy into media/ + add media row │ ├─ fablecut_analyze_reference(localPath) → analyze.js → blueprint + music media │ └─ fablecut_patch_project(ops[]) → project.json revision++ → Browser SSE sync

这条链路的妙处是所有 agent 写入都回到同一份 project.json，因此浏览器只要接收 SSE 并重新 applyProject 就能展示外部修改。服务端 watch 会广播见 sources/fablecut/server.js:61；浏览器 SSE 连接见 sources/fablecut/app.js:293；applyProject 清理 runtime见 sources/fablecut/app.js:239；MCP 写项目函数见 sources/fablecut/mcp-server.js:31；patch 写回触发 revision见 sources/fablecut/mcp-server.js:314。

▸set_project 与 stale revision
