1503function applyTransition(p, type, k, W, H, dir) {
1504  if (!(k > 0)) return;
1505  switch (type) {
1506    case "fade": case "dissolve":
1507      p.opacity *= 1 - k; p.volume *= 1 - k; break;
1508    case "slide-left":  p.x = (+p.x || 0) - dir * k * W; break;
1509    case "slide-right": p.x = (+p.x || 0) + dir * k * W; break;
1510    case "slide-up":    p.y = (+p.y || 0) - dir * k * H; break;
1511    case "slide-down":  p.y = (+p.y || 0) + dir * k * H; break;
1512    case "zoom":
1513      p.scale = (+p.scale || 1) * (1 - 0.6 * k); p.opacity *= 1 - k; break;
1514    case "wipe": case "wipe-left": p._wipe = k; p._wipeDir = "left"; break;
1515    case "wipe-right": p._wipe = k; p._wipeDir = "right"; break;
1516    case "wipe-up":    p._wipe = k; p._wipeDir = "up"; break;
1517    case "wipe-down":  p._wipe = k; p._wipeDir = "down"; break;
1518    case "iris": p._iris = k; break;
1519    case "spin":
1520      p.rotation = (+p.rotation || 0) + dir * k * 200;
1521      p.scale = (+p.scale || 1) * (1 - 0.4 * k);
1522      p.opacity *= 1 - k; break;
1523    case "blur":
1524      p.blur = (+p.blur || 0) + k * 24;
1525      p.opacity *= 1 - k * k; p.volume *= 1 - k; break;
1526    case "whip":
1527      p.x = (+p.x || 0) - dir * EASE["ease-in"](k) * W * 1.4;
1528      p.blur = (+p.blur || 0) + k * 16; break;
1529    case "glitch": { // RGB split + horizontal jitter, deterministic
1530      const j = Math.sin(k * 61.7) * Math.sin(k * 23.3);
1531      p.rgbSplit = (+p.rgbSplit || 0) + k * 14;

▸MediaRecorder fallback

当快导出不可用时，浏览器还有 MediaRecorder fallback；这说明项目对不同浏览器和缺 ffmpeg 环境有降级路径。fallback 标题见 sources/fablecut/app.js:2335；pickMime 候选见 sources/fablecut/app.js:2337；recorder 状态变量见 sources/fablecut/app.js:2336；server HAS_FFMPEG见 sources/fablecut/server.js:32；probe ffmpeg见 sources/fablecut/app.js:235。

  • 逐帧 seek 对长视频素材成本高，尤其是带 speed ramp 的 clip。
  • AI mask 在导出中同步刷新，质量稳定但增加 GPU/CPU 压力。
  • 音频混音依赖浏览器 OfflineAudioContext，复杂素材数量多时会占内存。
